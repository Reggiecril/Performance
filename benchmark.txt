
VM:
CPU:
sysbench cpu --cpu-max-prime=50000 --report-interval=1 run
Memory:
sysbench memory --memory-block-size=2M --memory-total-size=100T --report-interval=1 run
Thread:
sysbench threads --thread-yields=10000 --thread-locks=100 --report-interval=1 run
File:
sysbench  fileio --threads=16 --file-total-size=50G --file-test-mode=rndrw prepare
sysbench  fileio --threads=16 --file-total-size=20G --file-test-mode=rndrw --report-interval=5 run
sysbench  fileio --threads=16 --file-total-size=50G --file-test-mode=rndrw cleanup





Docker
Cpu:
docker exec -it  sysbench sysbench cpu --cpu-max-prime=50000000 --report-interval=1 run
Memory:
docker exec -it sysbench sysbench memory --memory-block-size=1M --memory-total-size=100G --num-threads=1  --report-interval=2  run
Thread:
docker exec -it sysbench sysbench threads --num-threads=500 --thread-yields=100 --thread-locks=4 --report-interval=5 run
File:
docker exec -it sysbench sysbench fileio --threads=20 --file-total-size=2G prepare
docker exec -it sysbench sysbench fileio --threads=20 --file-total-size=2G --file-test-mode=rndrw --report-interval=5 run
docker exec -it sysbench sysbench fileio --threads=20 --file-total-size=2G cleanup
